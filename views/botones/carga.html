function crono() {
     this.numid=0;
     this.empieza=function () {    
     this.numid=setInterval(function () {document.getElementById("m").value=document.getElementById("m").value+1;},1000); 
       }
       this.fin=function () {clearInterval(this.numid);}
     }
     window.onload=function () {
     var c = new crono();
     document.getElementById("empieza").onclick=function () {
     c.empieza();  
    }
    document.getElementById("fin").onclick=function () {
    c.fin();
    }
    }




  
   <button id="empieza">Start</button>
   <button id="fin">Stop</button>
   <meter id="m" value="0" max="50"></meter>
